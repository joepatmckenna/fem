    m = [len(np.unique(letter)) for letter in words]
    mm = np.insert(np.cumsum(m), 0, 0)
    M = np.sum(m)
    w, d = np.zeros((M, M)), []
    for i in range(words.shape[0]):
        not_i = np.delete(range(n), i)
        start = time.time()
        res = fem.discrete.fit.fit(words[not_i], words[[i]], overfit=False)
        end = time.time()
        print 'letter %i, fit time: %.02f' % (i+1, end-start)
        w[mm[i]:mm[i+1],:mm[i]] = res[0][1][:,:mm[i]]
        w[mm[i]:mm[i+1],mm[i+1]:] = res[0][1][:,mm[i]:]
        d.append(res[1][0])
    np.save(w_file, w)
    np.save(d_file, d)
